```javascript
this의 스코프
1. 전역에서 this > 브라우저 환경에서는 window 객체, node.js 환경에서는 global 객체

2. 객체의 메서드 내부에서의 this는 해당 메서드를 소유한 객체
 const obj = {
   name: 'test',
   some() {
   console.log(this.name); // 'test'
   }
 }

3. 일반 함수 내부에서의 this는 전역 객체
 function test() {
  console.log(this); // window
   function test2() {
     console.log(this); // window
   }
   test();
 }

4. 생성자 함수에서의 this는 새롭게 생성된 인스턴스
 function Person(name) {
   this.name = name;
 }
 const instance = new Person('Chris');
 console.log(person.name); // Chris

5. 화살표 함수의 this는 자기가 정의된 스코프에 존재하는 this를 가리킴
 const obj = {
   name: 'Alice',
   regular: function() {
     const someRegular = function () {
      console.log('regular', this); // window
     }
     const someArrow = () => {
      console.log('arrow', this); // obj
     }
     console.log(this); // obj
     someRegular();
     someArrow();
   },
   arrow: () => {
    console.log(this); // window
   },
   nested: {
     name: 'Chris',
     regular: function() {
      console.log(this); // nested
     },
     arrow: () => {
      console.log(this); // window
     }
   }
 }
 obj.regular();
 obj.arrow();
 obj.nested.regular();
 obj.nested.arrow();
```
